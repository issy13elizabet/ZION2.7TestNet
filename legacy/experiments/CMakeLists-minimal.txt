cmake_minimum_required(VERSION 3.12)
project(zion-minimal-ui)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# ZION Minimal UI Executable
add_executable(zion-minimal-ui
    src/main/zion-minimal-ui.cpp
)

# Windows-specific settings
if(WIN32)
    # Enable colored console output on Windows 10+
    target_compile_definitions(zion-minimal-ui PRIVATE 
        _WIN32_WINNT=0x0A00
        NOMINMAX
    )
    
    # Link Windows console libraries
    target_link_libraries(zion-minimal-ui PRIVATE
        kernel32
        user32
    )
endif()

# Set output directory
set_target_properties(zion-minimal-ui PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/bin"
)

# Compiler-specific flags
if(MSVC)
    target_compile_options(zion-minimal-ui PRIVATE /W3)
else()
    target_compile_options(zion-minimal-ui PRIVATE -Wall -Wextra)
endif()
